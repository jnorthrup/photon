88  public boolean novel(TermLink termLink, long currentTime) {
89  Term bTerm = termLink.getTerm();
90  if (bTerm.equals(getTargetTask().getSentence().getContent())) {
91  return false;
92  }
93  String linkKey = termLink.getKey();
94  int next = 0;
95  int i;
96  for (i = 0; i < getCounter(); i++) {
97  next = i % Parameters.TERM_LINK_RECORD_LENGTH;
98  if (linkKey.equals(this.getRecordedLinks().get(next))) {
99  if (currentTime < getRecordingTime().get(next) + Parameters.TERM_LINK_RECORD_LENGTH) {
100  return false;
101  } else {
102  getRecordingTime().set(next, currentTime);
103  return true;
104  }
105  }
106  }
107  next = i % Parameters.TERM_LINK_RECORD_LENGTH;
108  getRecordedLinks().add(next, linkKey); // add knowledge reference to recordedLinks
109  getRecordingTime().add(next, currentTime);
110  if (getCounter() < Parameters.TERM_LINK_RECORD_LENGTH) { // keep a constant length
111  setCounter(getCounter() + 1);
112  }
113  return true;
114  }

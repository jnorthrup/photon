233  private static BudgetValue budgetInference(float qual, int complexity,
234  WorkSpace memory) {
235  Item t = memory.getCurrentTaskLink();
236  if (t == null)
237  t = memory.getCurrentTask();
238  float priority = t.getPriority();
239  float durability = t.getDurability();
240  float quality = (float) (qual / Math.sqrt(complexity));
241  TermLink bLink = memory.getCurrentBeliefLink();
242  if (bLink != null) {
243  priority = aveAri(priority, bLink.getPriority());
244  durability = aveAri(durability, bLink.getDurability());
245  Item.incPriority(bLink.getBudget(), quality);
246  }
247  return new BudgetValue(and(priority, quality),
248  and(durability, quality), quality);
249  }

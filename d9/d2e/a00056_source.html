101  public boolean nextInput() {
102  if (timer > 0) {
103  timer--;
104  return true;
105  }
106  if (inExp == null) {
107  return false;
108  }
109  String line = null;
110  while (timer == 0) {
111  try {
112  line = inExp.readLine();
113  if (line == null) {
114  inExp.close();
115  inExp = null;
116  return false;
117  }
118  } catch (IOException ex) {
119  System.out.println("i/o error: " + ex.getMessage());
120  }
121  line = line.trim();
122  // read NARS language or an integer
123  if (line.length() > 0) {
124  try {
125  timer = Integer.parseInt(line);
126  ReasonerBatch.walk(reasoner, timer);
127  } catch (NumberFormatException e) {
128  ReasonerBatch.textInputLine(reasoner, line);
129  }
130  }
131  }
132  return true;

69  public static Statement make(String relation, Term subject, Term predicate, Memory memory) {
70  if (invalidStatement(subject, predicate)) {
71  return null;
72  }
73  if (relation.equals(Symbols.INHERITANCE_RELATION)) {
74  return Inheritance.make(subject, predicate, memory);
75  }
76  if (relation.equals(Symbols.SIMILARITY_RELATION)) {
77  return Similarity.make(subject, predicate, memory);
78  }
79  if (relation.equals(Symbols.INSTANCE_RELATION)) {
80  return Instance.make(subject, predicate, memory);
81  }
82  if (relation.equals(Symbols.PROPERTY_RELATION)) {
83  return Property.make(subject, predicate, memory);
84  }
85  if (relation.equals(Symbols.INSTANCE_PROPERTY_RELATION)) {
86  return InstanceProperty.make(subject, predicate, memory);
87  }
88  if (relation.equals(Symbols.IMPLICATION_RELATION)) {
89  return Implication.Companion.make(subject, predicate, memory);
90  }
91  if (relation.equals(Symbols.EQUIVALENCE_RELATION)) {
92  return Equivalence.Companion.make(subject, predicate, memory);
93  }
94  return null;

155  public static boolean invalidStatement(Term subject, Term predicate) {
156  if (subject.equals(predicate)) {
157  return true;
158  }
159  if ((subject instanceof CompoundTerm) && ((CompoundTerm) subject).containComponent(predicate)) {
160  return true;
161  }
162  if ((predicate instanceof CompoundTerm) && ((CompoundTerm) predicate).containComponent(subject)) {
163  return true;
164  }
165  if ((subject instanceof Statement) && (predicate instanceof Statement)) {
166  Statement s1 = (Statement) subject;
167  Statement s2 = (Statement) predicate;
168  Term t11 = s1.getSubject();
169  Term t12 = s1.getPredicate();
170  Term t21 = s2.getSubject();
171  Term t22 = s2.getPredicate();
172  if (t11.equals(t22) && t12.equals(t21)) {
173  return true;
174  }
175  }
176  return false;
177  }

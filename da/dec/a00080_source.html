75  public static Term make(List<Term> argList, Memory memory) {
76  Term r = null;
77  if (argList.size() >= 2) {
78  Term relation = argList.get(0);

80  int index = 0;

82  if (argList.get(j).getName().charAt(0) != Symbols.IMAGE_PLACE_HOLDER) {

102  public static Term make(Product product, Term relation, Integer index, Memory memory) {
103  Term r = null;
104 
105  e:
106  {
107  if (relation instanceof Product) {
108  Product p2 = (Product) relation;
109  if ((product.size() == 2) && (p2.size() == 2)) {
110  // (/,_,(*,a,b),b) is reduced to a
111  if ((index == 0) && product.componentAt(1).equals(p2.componentAt(1))) {
112  r = p2.componentAt(0);
113  break e;
114  } else // (/,(*,a,b),a,_) is reduced to b
115  if ((index == 1) && product.componentAt(0).equals(p2.componentAt(0))) {
116  r = p2.componentAt(1);
117  break e;
118  }
119  }
120  }
121  List<Term> argument = product.cloneComponents();
122  argument.set(index, relation);
123  r = make(argument, index, memory);
124  }

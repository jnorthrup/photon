157  public boolean putIn(Type newItem) {
158  String newKey = newItem.getKey();
159  Type oldItem = nameTable.put(newKey, newItem);
160  if (oldItem != null) { // merge duplications
161  outOfBase(oldItem);
162  Item.merge(newItem.getBudget(), oldItem);
163  }
164  Type overflowItem = intoBase(newItem); // put the (new or merged) item into itemTable
165  if (overflowItem != null) { // remove overflow
166  String overflowKey = overflowItem.getKey();
167  nameTable.remove(overflowKey);
168  return (overflowItem != newItem);
169  } else {
170  return true;

190  public Type takeOut() {
191  if (nameTable.isEmpty()) { // empty bag
192  return null;
193  }
194  if (emptyLevel(currentLevel) || (currentCounter == 0)) { // done with the current level
195  currentLevel = DISTRIBUTOR.pick(levelIndex);
196  levelIndex = DISTRIBUTOR.next(levelIndex);
197  while (emptyLevel(currentLevel)) { // look for a non-empty level
198  currentLevel = DISTRIBUTOR.pick(levelIndex);
199  levelIndex = DISTRIBUTOR.next(levelIndex);
200  }
201  if (currentLevel < THRESHOLD) { // for dormant levels, take one item
202  currentCounter = 1;
203  } else { // for active levels, take all current items
204  currentCounter = itemTable.get(currentLevel).size();
205  }
206  }
207  Type selected = takeOutFirst(currentLevel); // take out the first item in the level
208  currentCounter--;
209  nameTable.remove(selected.getKey());
210  refresh();
211  return selected;
212  }

253  private Type intoBase(Type newItem) {
254  Type oldItem = null;
255  int inLevel = getLevel(newItem);
256  if (nameTable.size() > capacity) { // the bag is full
257  int outLevel = 0;
258  while (emptyLevel(outLevel)) {
259  outLevel++;
260  }
261  if (outLevel > inLevel) { // ignore the item and exit
262  return newItem;
263  } else { // remove an old item in the lowest non-empty level
264  oldItem = takeOutFirst(outLevel);
265  }
266  }
267  itemTable.get(inLevel).add(newItem); // FIFO
268  mass += (inLevel + 1); // increase total mass
269  refresh(); // refresh the window
270  return oldItem; // TODO return null is a bad smell
271  }

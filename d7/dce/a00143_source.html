238  public static void activatedTask(Memory memory, BudgetValue budget, Sentence sentence, Sentence candidateBelief) {
239  Task task = new Task(sentence, budget, memory.getCurrentTask(), sentence, candidateBelief);
240  memory.getRecorder().append("!!! Activated: " + task + "\n");
241  if (sentence.isQuestion()) {
242  float s = task.getBudget().summary();
243 // float minSilent = reasoner.getMainWindow().silentW.value() / 100.0f;
244  float minSilent = ReasonerBatch.getSilenceValue(memory.getReasoner().getSilenceValue()).get() / 100.0f;
245  // only report significant derived Tasks
246  if (s > minSilent) report(memory, task.getSentence(), false);
247  }
248  memory.getNewTasks().add(task);

331  public static void singlePremiseTask(Memory memory, Term newContent, int punctuation, TruthValue newTruth, BudgetValue newBudget) {
332  Sentence taskSentence = memory.getCurrentTask().getSentence();
333  if (taskSentence.isJudgment() || null == memory.getCurrentBelief()) {
334  memory.setNewStamp(Stamp.createStamp(taskSentence.getStamp(), Memory.getTime(memory)));
335  } else {
336  memory.setNewStamp(Stamp.createStamp(memory.getCurrentBelief().getStamp(), Memory.getTime(memory)));
337  }
338  Sentence newSentence = new Sentence(newContent, punctuation, newTruth, memory.getNewStamp(), taskSentence.getRevisible());
339  Task newTask = new Task(newSentence, newBudget, memory.getCurrentTask(), null);
340  Memory.derivedTask(memory, newTask);
341  }

351  public static void workCycle(Memory memory, long clock) {
352  memory.getRecorder().append(" --- " + clock + " ---\n");
353  processNewTask(memory);
354  if (noResult(memory)) { // necessary?
355  processNovelTask(memory);
356  }
357  if (noResult(memory)) { // necessary?
358  processConcept(memory);
359  }
360  memory.getNovelTasks().refresh();
361  }

369  private static void processNewTask(Memory memory) {
370  Task task;
371  int counter = memory.getNewTasks().size(); // don't include new tasks produced in the current workCycle
372  while (0 < counter--) {
373  task = memory.getNewTasks().remove(0);
374  if (task.isInput() || null != termToConcept(memory, task.getContent())) { // new input or existing concept
375  immediateProcess(memory, task);
376  } else {
377  Sentence s = task.getSentence();
378  if (s.isJudgment()) {
379  double d = s.getTruth().getExpectation();
380  if (Parameters.DEFAULT_CREATION_EXPECTATION < d)
381  memory.getNovelTasks().putIn(task); // new concept formation
382  else {
383  memory.getRecorder().append("!!! Neglected: " + task + "\n");
384  }
385  }
386  }
387  }
388  }

405  private static void processConcept(Memory memory) {
406  memory.setCurrentConcept(memory.getConcepts().takeOut());
407  if (null != memory.getCurrentConcept()) {
408  memory.setCurrentTerm(memory.getCurrentConcept().getTerm());
409  memory.getRecorder().append(" * Selected Concept: " + memory.getCurrentTerm() + "\n");
410  memory.getConcepts().putBack(memory.getCurrentConcept()); // current Concept remains in the bag all the time
411  Concept.fireFromProcessConcept(memory.getCurrentConcept().getMemory(), memory.getCurrentConcept().getTaskLinks(), memory.getCurrentConcept().getTermLinks()); // a working workCycle
412  }

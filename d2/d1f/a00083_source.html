79  public static Term make(List<Term> argList, Memory memory) {
80  if (argList.size() < 2) {
81  return null;
82  }
83  Term relation = argList.get(0);
84  List<Term> argument = new ArrayList<Term>();
85  int index = 0;
86  for (int j = 1; j < argList.size(); j++) {
87  if (argList.get(j).getName().charAt(0) == Symbols.IMAGE_PLACE_HOLDER) {
88  index = j - 1;
89  argument.add(relation);
90  } else {
91  argument.add(argList.get(j));
92  }
93  }
94  return make(argument, index, memory);
95  }

105  public static Term make(Product product, Term relation, Integer index, Memory memory) {
106  if (relation instanceof Product) {
107  Product p2 = (Product) relation;
108  if (product.size() == 2 && p2.size() == 2) {
109  if (index == 0 && product.componentAt(1).equals(p2.componentAt(1))) {// (\,_,(*,a,b),b) is reduced to a
110  return p2.componentAt(0);
111  }
112  if (index == 1 && product.componentAt(0).equals(p2.componentAt(0))) {// (\,(*,a,b),a,_) is reduced to b
113  return p2.componentAt(1);
114  }
115  }
116  }
117  List<Term> argument = product.cloneComponents();
118  argument.set(index, relation);
119  return make(argument, index, memory);
120  }

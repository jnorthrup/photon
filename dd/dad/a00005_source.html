133  private void processJudgment(Task task) {
134  Sentence judg = task.getSentence();
135  Sentence oldBelief = evaluation(judg, getBeliefs());
136  if (oldBelief != null) {
137  Stamp newStamp = judg.getStamp(), oldStamp = oldBelief.getStamp();
138  if (newStamp.equals(oldStamp)) {
139  task.getBudget().decPriority(0); // duplicated task
140  return;
141  } else if (LocalRules.revisible(judg, oldBelief)) {
142  getMemory().setNewStamp(Stamp.make(newStamp, oldStamp, Memory.getTime(getMemory())));
143  if (getMemory().getNewStamp() != null) {
144  getMemory().setCurrentBelief(oldBelief);
145  LocalRules.revision(judg, oldBelief, false, getMemory());
146  }
147  }
148  }
149  if (task.getBudget().aboveThreshold()) {
150  for (Task ques : getQuestions()) {
151 // LocalRules.trySolution(ques.getSentence(), judg, ques, memory);
152  LocalRules.trySolution(judg, ques, getMemory());
153  }
154  addToTable(judg, getBeliefs(), Parameters.MAXIMUM_BELIEF_LENGTH);
155  }
156  }

164  public float processQuestion(Task task) {
165  Sentence ques = task.getSentence();
166  boolean newQuestion = true;
167  if (getQuestions() != null) {
168  for (Task t : getQuestions()) {
169  Sentence q = t.getSentence();
170  if (q.getContent().equals(ques.getContent())) {
171  ques = q;
172  newQuestion = false;
173  break;
174  }
175  }
176  if (newQuestion) {
177  getQuestions().add(task);
178  }
179  if (getQuestions().size() > Parameters.MAXIMUM_QUESTIONS_LENGTH) {
180  getQuestions().remove(0); // FIFO
181  }
182  Sentence newAnswer = evaluation(ques, getBeliefs());
183  if (newAnswer != null) {
184 // LocalRules.trySolution(ques, newAnswer, task, memory);
185  LocalRules.trySolution(newAnswer, task, getMemory());
186  return newAnswer.getTruth().getExpectation();
187  } else {
188  return 0.5f;
189  }
190  }
191  throw new Error("this state was not reachable from previous code");
192  }

202  private void linkToTask(Task task) {
203  BudgetValue taskBudget = task.getBudget();
204  TaskLink taskLink = new TaskLink(task, null, taskBudget); // link type: SELF
205  insertTaskLink(taskLink);
206  if (getTerm() instanceof CompoundTerm) {
207  if (getTermLinkTemplates().size() > 0) {
208  BudgetValue subBudget = distributeAmongLinks(taskBudget, getTermLinkTemplates().size());
209  if (subBudget.aboveThreshold()) {
210  Term componentTerm;
211  Concept componentConcept;
212  for (TermLink termLink : getTermLinkTemplates()) {
213 // if (!(task.isStructural() && (termLink.getType() == TermLink.TRANSFORM))) { // avoid circular transform
214  taskLink = new TaskLink(task, termLink, subBudget);
215  componentTerm = termLink.getTerm();
216  componentConcept = Memory.getConcept(getMemory(), componentTerm);
217  if (componentConcept != null) {
218  componentConcept.insertTaskLink(taskLink);
219  }
220 // }
221  }

223  }

236  private static void addToTable(Sentence newSentence, List<Sentence> table, int capacity) {
237  float rank1 = rankBelief(newSentence); // for the new isBelief
238  Sentence judgment2;
239  float rank2;
240  int i;
241  for (i = 0; i < table.size(); i++) {
242  judgment2 = table.get(i);
243  rank2 = rankBelief(judgment2);
244  if (rank1 >= rank2) {
245  if (newSentence.equivalentTo(judgment2)) {
246  return;
247  }
248  table.add(i, newSentence);
249  break;
250  }
251  }
252  if (table.size() >= capacity) {
253  while (table.size() > capacity) {
254  table.remove(table.size() - 1);
255  }
256  } else if (i == table.size()) {
257  table.add(newSentence);
258  }
259  }

268  private static Sentence evaluation(SentenceStruct query, Iterable<Sentence> list) {
269  if (list == null) {
270  return null;
271  }
272  float currentBest = 0;
273  float beliefQuality;
274  Sentence candidate = null;
275  for (Sentence judg : list) {
276  beliefQuality = LocalRules.solutionQuality(query, judg);
277  if (beliefQuality > currentBest) {
278  currentBest = beliefQuality;
279  candidate = judg;
280  }
281  }
282  return candidate;
283  }

307  public void buildTermLinks(BudgetStruct taskBudget) {
308  Term t;
309  Concept concept;
310  TermLink termLink1, termLink2;
311  if (getTermLinkTemplates().size() > 0) {
312  BudgetValue subBudget = distributeAmongLinks(taskBudget, getTermLinkTemplates().size());
313  if (subBudget.aboveThreshold()) {
314  for (TermLink template : getTermLinkTemplates()) {
315  if (template.getType() != TermLink.TRANSFORM) {
316  t = template.getTerm();
317  concept = Memory.getConcept(getMemory(), t);
318  if (concept != null) {
319  termLink1 = new TermLink(t, template, subBudget);
320  insertTermLink(termLink1); // this termLink to that
321  termLink2 = new TermLink(getTerm(), template, subBudget);
322  concept.insertTermLink(termLink2); // that termLink to this
323  if (t instanceof CompoundTerm) {
324  concept.buildTermLinks(subBudget);
325  }
326  }
327  }
328  }
329  }
330  }

446  public static void fireFromProcessConcept(Memory memory, TaskLinkBag taskLinks, TermLinkBag termLinks) {
447  TaskLink currentTaskLink = taskLinks.takeOut();
448  if (currentTaskLink != null) {
449  memory.setCurrentTaskLink(currentTaskLink);
450  memory.setCurrentBeliefLink(null);
451  memory.getRecorder().append(" * Selected TaskLink: " + currentTaskLink + "\n");
452  Task task = currentTaskLink.getTargetTask();
453  memory.setCurrentTask(task); // one of the two places where this variable is set
454  if (currentTaskLink.getType() == TermLink.TRANSFORM) {
455  RuleTables.transformTask(currentTaskLink, memory); // to turn this into structural inference as below?
456  }
457  int termLinkCount = Parameters.MAX_REASONED_TERM_LINK;
458  while (Memory.noResult(memory) && (termLinkCount > 0)) {
459  TermLink termLink = termLinks.takeOut(currentTaskLink, Memory.getTime(memory));
460  if (termLink != null) {
461  memory.getRecorder().append(" * Selected TermLink: " + termLink + "\n");
462  memory.setCurrentBeliefLink(termLink);
463  RuleTables.reason(currentTaskLink, termLink, memory);
464  termLinks.putBack(termLink);
465  termLinkCount--;
466  } else {
467  termLinkCount = 0;
468  }
469  }
470  taskLinks.putBack(currentTaskLink);
471  }
472  }

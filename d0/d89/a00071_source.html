78  public static Task parseTask(String s, Memory memory, long time) {
79  StringBuffer buffer = new StringBuffer(s);
80  Task task = null;
81  try {
82  String budgetString = getBudgetString(buffer);
83  String truthString = getTruthString(buffer);
84  String str = buffer.toString().trim();
85  int last = str.length() - 1;
86  char punc = str.charAt(last);
87  Stamp stamp = Stamp.createStamp(time);
88  TruthValue truth = parseTruth(truthString, punc);
89  Term content = parseTerm(str.substring(0, last), memory);
90  Sentence sentence = new Sentence(content, punc, truth, stamp);
91  if (content instanceof Conjunction && Variable.containVarDep(content.getName()))
92  sentence.setRevisible(false);
93  BudgetValue budget = parseBudget(budgetString, punc, truth);
94  task = new Task(sentence, budget);
95  } catch (InvalidInputException e) {
96  String message = " !!! INVALID INPUT: parseTask: " + buffer + " --- " + e.getMessage();
97  System.out.println(message);
98  showWarning(message);
99  }
100  return task;
101  }

179  private static BudgetValue parseBudget(String s, char punctuation, TruthHandle truth) throws InvalidInputException {
180  float priority, durability;
181  switch (punctuation) {
182  case Symbols.JUDGMENT_MARK:
183  priority = Parameters.DEFAULT_JUDGMENT_PRIORITY;
184  durability = Parameters.DEFAULT_JUDGMENT_DURABILITY;
185  break;
186  case Symbols.QUESTION_MARK:
187  priority = Parameters.DEFAULT_QUESTION_PRIORITY;
188  durability = Parameters.DEFAULT_QUESTION_DURABILITY;
189  break;
190  default:
191  throw new InvalidInputException("unknown punctuation: '" + punctuation + "'");
192  }
193  if (null != s) { // overrite default
194  int i = s.indexOf(Symbols.VALUE_SEPARATOR);
195  if (0 > i) { // default durability
196  priority = Float.parseFloat(s);
197  } else {
198  priority = Float.parseFloat(s.substring(0, i));
199  durability = Float.parseFloat(s.substring(i + 1));
200  }
201  }
202  float quality = null == truth ? 1 : BudgetFunctions.truthToQuality(truth);
203  return new BudgetValue(priority, durability, quality);
204  }

220  public static Term parseTerm(String s0, Memory memory) {
221  String s = s0.trim();
222  try {
223  assert !s.isEmpty() : "missing content";
224  Term t = Memory.nameToListedTerm(memory, s); // existing constant or operator
225  if (null == t) {
226  int index = s.length() - 1;
227  char first = s.charAt(0);
228  char last = s.charAt(index);
229  switch (first) {
230  case Symbols.COMPOUND_TERM_OPENER:
231  assert Symbols.COMPOUND_TERM_CLOSER == last : "missing CompoundTerm closer";
232  return parseCompoundTerm(s.substring(1, index), memory);
233  case Symbols.SET_EXT_OPENER:
234  assert Symbols.SET_EXT_CLOSER == last : "missing ExtensionSet closer";
235  return SetExt.make(parseArguments(s.substring(1, index) + Symbols.ARGUMENT_SEPARATOR, memory), memory);
236  case Symbols.SET_INT_OPENER:
237  assert Symbols.SET_INT_CLOSER == last : "missing IntensionSet closer";
238  return SetInt.make(parseArguments(s.substring(1, index) + Symbols.ARGUMENT_SEPARATOR, memory), memory);
239  case Symbols.STATEMENT_OPENER:
240  assert Symbols.STATEMENT_CLOSER == last : "missing Statement closer";
241  return parseStatement(s.substring(1, index), memory);
242  default:
243  return parseAtomicTerm(s);
244  }
245  }
246  return t;
247  } catch (InvalidInputException e) {
248  String message = " !!! INVALID INPUT: parseTerm: " + s + " --- " + e.getMessage();
249  System.out.println(message);
250  showWarning(message);
251  }
252  return null;
253  }

287  private static Statement parseStatement(String s0, Memory memory) throws InvalidInputException {
288  String s = s0.trim();
289  int i = topRelation(s);
290  if (0 > i) {
291  throw new InvalidInputException("invalid statement");
292  }

295  Term predicate = parseTerm(s.substring(i + 3), memory);
296  Statement t = Statement.make(relation, subject, predicate, memory);
297  assert null != t : "invalid statement";
298  return t;

309  String s = s0.trim();

330  private static List<Term> parseArguments(String s0, Memory memory) throws InvalidInputException {
331  String s = s0.trim();
332  List<Term> list = new ArrayList<Term>();
333  int start = 0;
334  int end = 0;
335  Term t;
336  while (end < s.length() - 1) {
337  end = nextSeparator(s, start);
338  t = parseTerm(s.substring(start, end), memory); // recursive call
339  list.add(t);
340  start = end + 1;
341  }
342  if (list.isEmpty()) {
343  throw new InvalidInputException("null argument");
344  }
345  return list;
346  }

355  private static int nextSeparator(String s, int first) {
356  int levelCounter = 0;
357  int i = first;
358  while (i < s.length() - 1) {
359  if (isOpener(s, i)) {
360  levelCounter++;
361  } else if (isCloser(s, i)) {
362  levelCounter--;
363  } else if (Symbols.ARGUMENT_SEPARATOR == s.charAt(i)) {
364  if (0 == levelCounter) {
365  break;
366  }
367  }
368  i++;
369  }
370  return i;
371  }

381  private static int topRelation(String s) { // need efficiency improvement
382  int levelCounter = 0;
383  int i = 0;
384  while (i < s.length() - 3) { // don't need to check the last 3 characters
385  if (0 == levelCounter && isRelation(s.substring(i, i + 3))) {
386  return i;
387  }
388  if (isOpener(s, i)) {
389  levelCounter++;
390  } else if (isCloser(s, i)) {
391  levelCounter--;
392  }
393  i++;
394  }
395  return -1;

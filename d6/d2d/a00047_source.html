53  static void analogy(Term term1, Term term2, Sentence asym,
54  SentenceStruct sym, int figure, Memory memory) {

56  return;
57  }
58  Statement asymSt = (Statement) asym.getContent();
59  // Statement symSt = (Statement) sym.getContent();
60  TruthValue truth = null;
61  BudgetValue budget;
62  Sentence sentence = memory.getCurrentTask().getSentence();
63  CompoundTerm taskTerm = (CompoundTerm) sentence.getContent();
64  if (sentence.isQuestion()) {
65  if (taskTerm.isCommutative()) {
66  budget = BudgetFunctions.backwardWeak(memory, asym.getTruth());
67  } else {
68  budget = BudgetFunctions.backward(memory, sym.getTruth());
69  }
70  } else {

72  budget = BudgetFunctions.forward(memory, truth);
73  }
74  Term content = Statement.make(asymSt, term1, term2, memory);
75  Memory.doublePremiseTask(memory, content, truth, budget);

48  public static Task parseExperience(StringBuffer buffer, Memory memory, long time) {
49  int i = buffer.indexOf(Symbols.PREFIX_MARK + "");
50  if (0 < i) {
51  String prefix = buffer.substring(0, i).trim();
52  if (prefix.equals(Symbols.OUTPUT_LINE)) {
53  return null;
54  } else if (prefix.equals(Symbols.INPUT_LINE)) {
55  buffer.delete(0, i + 1);
56  }
57  }
58  char c = buffer.charAt(buffer.length() - 1);
59  if (Symbols.STAMP_CLOSER == c) {
60  int j = buffer.lastIndexOf(Symbols.STAMP_OPENER + "");
61  buffer.delete(j - 1, buffer.length());
62  }
63  return StringParser.parseTask(buffer.toString().trim(), memory, time);
64  }

158  public static void doTick(ReasonerBatch reasonerBatch) {
159  if (isDEBUG() && (reasonerBatch.isRunning() || 0 < reasonerBatch.getWalkingSteps() || !reasonerBatch.isFinishedInputs())) {
160  System.out.println("// doTick: "
161  + "walkingSteps " + reasonerBatch.getWalkingSteps()
162  + ", clock " + reasonerBatch.getClock()
163  + ", getTimer " + reasonerBatch.getTimer()
164  + "\n// memory.getExportStrings() " + reasonerBatch.getMemory().getExportStrings()
165  );
166  System.out.flush();
167  }
168  switch (reasonerBatch.getWalkingSteps()) {
169  case 0:
170  boolean reasonerShouldRun = false;
171  for (InputChannel channelIn : reasonerBatch.getInputChannels()) {
172  reasonerShouldRun = reasonerShouldRun ||
173  channelIn.nextInput();
174  }
175  reasonerBatch.setFinishedInputs(!reasonerShouldRun);
176  break;
177  }
178  // forward to output Channels
179  List<String> output = reasonerBatch.getMemory().getExportStrings();
180  if (!output.isEmpty()) {
181  for (OutputChannel channelOut : reasonerBatch.getOutputChannels()) {
182  channelOut.nextOutput(output);
183  }
184  output.clear(); // this will trigger display the current value of timer in Memory.report()
185  }
186  if (reasonerBatch.isRunning() || 0 < reasonerBatch.getWalkingSteps()) {
187  reasonerBatch.setClock(reasonerBatch.getClock() + 1);
188  ReasonerBatch.tickTimer(reasonerBatch);
189  Memory.workCycle(reasonerBatch.getMemory(), reasonerBatch.getClock());
190  if (0 < reasonerBatch.getWalkingSteps()) {
191  reasonerBatch.setWalkingSteps(reasonerBatch.getWalkingSteps() - 1);
192  }
193  }
194  }

212  public static void textInputLine(ReasonerBatch reasonerBatch, String text) {
213  if (!text.isEmpty()) {
214  char c = text.charAt(0);
215  switch (c) {

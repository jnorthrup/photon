148  public static Term make(String op, List<Term> arg, Memory memory) {
149  if (1 == op.length()) {
150  if (Symbols.SET_EXT_OPENER == op.charAt(0)) {
151  return SetExt.make(arg, memory);
152  }
153  if (Symbols.SET_INT_OPENER == op.charAt(0)) {
154  return SetInt.make(arg, memory);
155  }
156  if (op.equals(Symbols.INTERSECTION_EXT_OPERATOR)) {
157  return IntersectionExt.make(arg, memory);
158  }
159  if (op.equals(Symbols.INTERSECTION_INT_OPERATOR)) {
160  return IntersectionInt.make(arg, memory);
161  }
162  if (op.equals(Symbols.DIFFERENCE_EXT_OPERATOR)) {
163  return DifferenceExt.Companion.make(arg, memory);
164  }
165  if (op.equals(Symbols.DIFFERENCE_INT_OPERATOR)) {
166  return DifferenceInt.Companion.make(arg, memory);
167  }
168  if (op.equals(Symbols.PRODUCT_OPERATOR)) {
169  return Product.make(arg, memory);
170  }
171  if (op.equals(Symbols.IMAGE_EXT_OPERATOR)) {
172  return ImageExt.make(arg, memory);
173  }
174  if (op.equals(Symbols.IMAGE_INT_OPERATOR)) {
175  return ImageInt.make(arg, memory);
176  }
177  }
178  if (2 == op.length()) {
179  if (op.equals(Symbols.NEGATION_OPERATOR)) {
180  return Negation.make(arg, memory);
181  }
182  if (op.equals(Symbols.DISJUNCTION_OPERATOR)) {
183  return Disjunction.Companion.make(arg, memory);
184  }
185  if (op.equals(Symbols.CONJUNCTION_OPERATOR)) {
186  return Conjunction.Companion.make(arg, memory);
187  }
188  }
189  return null;
190  }

476  public void renameVariables(HashMap<Variable, Variable> map) {
477  if (containVar()) {
478  for (int i = 0; i < components.size(); i++) {
479  Term term = componentAt(i);
480  if (term instanceof Variable) {
481  Variable var;
482  if (1 == term.getName().length()) { // anonymous variable from input
483  var = new Variable(term.getName().charAt(0) + "" + (map.size() + 1));
484  } else {
485  var = map.get(term);
486  if (null == var) {
487  var = new Variable(term.getName().charAt(0) + "" + (map.size() + 1));
488  }
489  }
490  if (!term.equals(var)) {
491  components.set(i, var);
492  }
493  map.put((Variable) term, var);
494  } else if (term instanceof CompoundTerm) {
495  ((CompoundTerm) term).renameVariables(map);
496  term.setName(((CompoundTerm) term).makeName());

547  private void prepareComponentLinks(ArrayList<TermLink> componentLinks, short type, CompoundTerm term) {
548  Term t1, t2, t3; // components at different levels
549  for (int i = 0; i < term.size(); i++) { // first level components
550  t1 = term.componentAt(i);
551  if (t1.isConstant()) {
552  componentLinks.add(new TermLink(t1, type, i));
553  }
554  if ((t1 instanceof Conjunction) && ((this instanceof Equivalence) || ((this instanceof Implication) && (0 == i)))) {
555  ((CompoundTerm) t1).prepareComponentLinks(componentLinks, TermLink.COMPOUND_CONDITION, (CompoundTerm) t1);
556  } else if (t1 instanceof CompoundTerm) {
557  for (int j = 0; j < ((CompoundTerm) t1).size(); j++) { // second level components
558  t2 = ((CompoundTerm) t1).componentAt(j);
559  if (t2.isConstant()) {
560  if ((t1 instanceof Product) || (t1 instanceof ImageExt) || (t1 instanceof ImageInt)) {
561  if (TermLink.COMPOUND_CONDITION == type) {
562  componentLinks.add(new TermLink(t2, TermLink.TRANSFORM, 0, i, j));
563  } else {
564  componentLinks.add(new TermLink(t2, TermLink.TRANSFORM, i, j));
565  }
566  } else {
567  componentLinks.add(new TermLink(t2, type, i, j));
568  }
569  }
570  if ((t2 instanceof Product) || (t2 instanceof ImageExt) || (t2 instanceof ImageInt)) {
571  for (int k = 0; k < ((CompoundTerm) t2).size(); k++) {
572  t3 = ((CompoundTerm) t2).componentAt(k);
573  if (t3.isConstant()) { // third level
574  if (TermLink.COMPOUND_CONDITION == type) {
575  componentLinks.add(new TermLink(t3, TermLink.TRANSFORM, 0, i, j, k));
576  } else {
577  componentLinks.add(new TermLink(t3, TermLink.TRANSFORM, i, j, k));
578  }
579  }
580  }
581  }
582  }
583  }
584  }

73  public static Term make(Term term1, Term term2, Memory memory) {
74  TreeSet<Term> set;
75  if ((term1 instanceof SetExt) && (term2 instanceof SetExt)) {
76  set = new TreeSet<Term>(((CompoundTerm) term1).cloneComponents());
77  set.addAll(((CompoundTerm) term2).cloneComponents()); // set union
78  return SetExt.make(set, memory);
79  }
80  if ((term1 instanceof SetInt) && (term2 instanceof SetInt)) {
81  set = new TreeSet<Term>(((CompoundTerm) term1).cloneComponents());
82  set.retainAll(((CompoundTerm) term2).cloneComponents()); // set intersection
83  return SetInt.make(set, memory);
84  }
85  if (term1 instanceof IntersectionInt) {
86  set = new TreeSet<Term>(((CompoundTerm) term1).cloneComponents());
87  if (term2 instanceof IntersectionInt) {
88  set.addAll(((CompoundTerm) term2).cloneComponents());
89  } // (|,(|,P,Q),(|,R,S)) = (|,P,Q,R,S)
90  else {
91  set.add((Term) term2.clone());
92  } // (|,(|,P,Q),R) = (|,P,Q,R)
93  } else if (term2 instanceof IntersectionInt) {
94  set = new TreeSet<Term>(((CompoundTerm) term2).cloneComponents());
95  set.add((Term) term1.clone()); // (|,R,(|,P,Q)) = (|,P,Q,R)
96  } else {
97  set = new TreeSet<Term>();
98  set.add((Term) term1.clone());
99  set.add((Term) term2.clone());
100  }
101  return make(set, memory);
102  }

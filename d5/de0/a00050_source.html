222  float f1 = v1.getFrequency();

224  float c1 = v1.getConfidence();
225  float c2 = v2.getConfidence();
226  float f0 = or(f1, f2);
227  float f = (f0 == 0) ? 0 : (and(f1, f2) / f0);

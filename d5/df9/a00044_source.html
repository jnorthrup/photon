56  static void structuralCompose2(CompoundTerm compound, Integer index, Statement statement, Integer side, Memory memory) {
57  if (compound.equals(statement.componentAt(side))) {
58  return;
59  }
60  Term sub = statement.getSubject();
61  Term pred = statement.getPredicate();
62  List<Term> components = compound.cloneComponents();
63  if ((0 == side && components.contains(pred)) || (1 == side && components.contains(sub))) {
64  return;
65  }
66  if (0 == side) {
67  if (!components.contains(sub)) {
68  if (components.contains(pred)) {
69  if (sub instanceof CompoundTerm) {
70  return;
71  } else {
72  components.set(index, sub);
73  sub = CompoundTerm.make(compound, components, memory);
74  pred = compound;
75  }
76  } else {
77  if (pred instanceof CompoundTerm) {
78  return;
79  } else {
80  sub = compound;
81  components.set(index, pred);
82  pred = CompoundTerm.make(compound, components, memory);
83  }
84  }
85  }
86  }
87  if (null != sub && null != pred) {
88  Term content = switchOrder(compound, index) ? Statement.make(statement, pred, sub, memory) : Statement.make(statement, sub, pred, memory);
89  if (null != content) {
90  Task task = memory.getCurrentTask();
91  Sentence sentence = task.getSentence();
92  TruthValue truth = sentence.getTruth();
93  BudgetValue budget;
94  if (sentence.isQuestion()) {
95  budget = BudgetFunctions.compoundBackwardWeak(content, memory);
96  } else {
97  if (1 < compound.size()) {
98  if (sentence.isJudgment()) {
99  truth = TruthFunctions.deduction(truth, RELIANCE);
100  } else {
101  return;
102  }
103  }
104  budget = compoundForward(memory, truth, content);
105  }
106  Memory.singlePremiseTask(memory, content, truth, budget);
107  }
108  }

117  static void structuralDecompose2(Statement statement, Memory memory) {
118  Term subj = statement.getSubject();
119  Term pred = statement.getPredicate();
120  if (subj.getClass() != pred.getClass()) {
121  return;
122  }
123  CompoundTerm sub = (CompoundTerm) subj;
124  CompoundTerm pre = (CompoundTerm) pred;
125  if (sub.size() != pre.size()) {
126  return;
127  }
128  int index = -1;
129  Term t1, t2;
130  for (int i = 0; i < sub.size(); i++) {
131  t1 = sub.componentAt(i);
132  t2 = pre.componentAt(i);
133  if (!t1.equals(t2)) {
134  if (0 > index) {
135  index = i;
136  } else {
137  return;
138  }
139  }
140  }
141  t1 = sub.componentAt(index);
142  t2 = pre.componentAt(index);
143  Term content;
144  if (switchOrder(sub, index)) {
145  content = Statement.make(statement, t2, t1, memory);
146  } else {
147  content = Statement.make(statement, t1, t2, memory);
148  }
149  if (null == content) {
150  return;
151  }
152  Task task = memory.getCurrentTask();
153  Sentence sentence = task.getSentence();
154  TruthValue truth = sentence.getTruth();
155  BudgetValue budget;
156  if (sentence.isQuestion()) {
157  budget = compoundBackward(content, memory);
158  } else {
159  if (1 < sub.size() && sentence.isJudgment()) {
160  return;
161 
162  }
163  budget = compoundForward(memory, truth, content);
164  }
165  Memory.singlePremiseTask(memory, content, truth, budget);

299  static void transformProductImage(Inheritance inh, CompoundTerm oldContent, List<Integer> indices, Memory memory) {
300  Term subject = inh.getSubject();
301  Term predicate = inh.getPredicate();
302  if (inh.equals(oldContent)) {
303  if (subject instanceof CompoundTerm) transformSubjectPI((CompoundTerm) subject, predicate, memory);
304  if (predicate instanceof CompoundTerm) transformPredicatePI(subject, (CompoundTerm) predicate, memory);
305  } else {
306  Integer index = indices.get(indices.size() - 1);
307  Integer side = indices.get(indices.size() - 2);
308  CompoundTerm comp = (CompoundTerm) inh.componentAt(side);
309  if ((comp instanceof Product)) {
310  if (0 == side) {
311  subject = comp.componentAt(index);
312  predicate = ImageExt.make((Product) comp, inh.getPredicate(), index, memory);
313 
314  } else {
315  subject = ImageInt.make((Product) comp, inh.getSubject(), index, memory);
316  predicate = comp.componentAt(index);
317 
318  }
319  } else if ((comp instanceof ImageExt) && 1 == side) {
320  if (index == ((ImageExt) comp).getRelationIndex()) {
321  subject = Product.make(comp, inh.getSubject(), index, memory);
322  predicate = comp.componentAt(index);
323  } else {
324  subject = comp.componentAt(index);
325  predicate = ImageExt.make((ImageExt) comp, inh.getSubject(), index, memory);
326  }
327  } else {
328  if (comp instanceof ImageInt && 0 == side) if (index != ((ImageInt) comp).getRelationIndex()) {
329  subject = ImageInt.make((ImageInt) comp, inh.getPredicate(), index, memory);
330  predicate = comp.componentAt(index);
331  } else {
332  subject = comp.componentAt(index);
333  predicate = Product.make(comp, inh.getPredicate(), index, memory);
334  }
335  else {
336  return;
337  }
338  }
339  Inheritance newInh = Inheritance.make(subject, predicate, memory);
340  Term content = null;
341  if (2 != indices.size()) if (!(oldContent instanceof Statement) || 1 != indices.get(0)) {
342  List<Term> componentList;
343  Term condition = oldContent.componentAt(0);
344  if ((oldContent instanceof Implication || (oldContent instanceof Equivalence))) {
345  if (!(condition instanceof Conjunction)) {
346  componentList = oldContent.cloneComponents();
347  componentList.set(indices.get(0), newInh);
348  content = Statement.make((Statement) oldContent, componentList.get(0), componentList.get(1), memory);
349  } else {
350  componentList = ((CompoundTerm) condition).cloneComponents();
351  componentList.set(indices.get(1), newInh);
352  Term newCond = CompoundTerm.make((CompoundTerm) condition, componentList, memory);
353  content = Statement.make((Statement) oldContent, newCond, ((Statement) oldContent).getPredicate(), memory);
354  }
355  } else {
356  componentList = oldContent.cloneComponents();
357  componentList.set(indices.get(0), newInh);
358  if (oldContent instanceof Conjunction) {
359  content = CompoundTerm.make(oldContent, componentList, memory);
360  }
361  }
362  } else {
363  content = Statement.make((Statement) oldContent, oldContent.componentAt(0), newInh, memory);
364  }
365  else {
366  content = newInh;
367  }
368  if (null != content) {
369  Sentence sentence = memory.getCurrentTask().getSentence();
370  TruthValue truth = sentence.getTruth();
371  BudgetValue budget;
372  budget = sentence.isQuestion() ? compoundBackward(content, memory) : compoundForward(memory, truth, content);
373  Memory.singlePremiseTask(memory, content, truth, budget);
374  }
375  }

388  private static void transformSubjectPI(CompoundTerm subject,
389  Term predicate, Memory memory) {
390  TruthValue truth = memory.getCurrentTask().getSentence().getTruth();
391  BudgetValue budget;
392  Inheritance inheritance;
393  Term newSubj, newPred;
394  if (subject instanceof Product) {
395  Product product = (Product) subject;
396  for (Integer i = 0; i < product.size(); i++) {
397  newSubj = product.componentAt(i);
398  newPred = ImageExt.make(product, predicate, i, memory);
399  inheritance = Inheritance.make(newSubj, newPred, memory);
400  budget = null == truth ? compoundBackward(inheritance,
401  memory) : compoundForward(memory, truth,
402  inheritance);
403  Memory.singlePremiseTask(memory, inheritance, truth, budget);
404  }
405  } else {
406  if (!(subject instanceof ImageInt)) {

408  }

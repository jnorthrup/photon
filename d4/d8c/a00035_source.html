51  static void composeCompound(Statement taskContent, Statement beliefContent,
52  int index, Memory memory) {
53  if (!memory.getCurrentTask().getSentence().isJudgment()
54  || taskContent.getClass() != beliefContent.getClass()) {
55  return;
56  }
57  Term componentT = taskContent.componentAt(1 - index);
58  Term componentB = beliefContent.componentAt(1 - index);
59  Term componentCommon = taskContent.componentAt(index);
60  if (componentT instanceof CompoundTerm
61  && ((CompoundTerm) componentT).containAllComponents(componentB)) {
62  decomposeCompound((CompoundTerm) componentT, componentB,
63  componentCommon, index, true, memory);
64  return;
65  } else if (componentB instanceof CompoundTerm
66  && ((CompoundTerm) componentB).containAllComponents(componentT)) {
67  decomposeCompound((CompoundTerm) componentB, componentT,
68  componentCommon, index, false, memory);
69  return;
70  }
71  TruthValue truthT = memory.getCurrentTask().getSentence().getTruth();
72  TruthValue truthB = memory.getCurrentBelief().getTruth();
73  TruthValue truthOr = TruthFunctions.union(truthT, truthB);
74  TruthValue truthAnd = TruthFunctions.intersection(truthT, truthB);
75  Term termOr = null;
76  Term termAnd = null;
77  if (index == 0) {
78  if (taskContent instanceof Inheritance) {
79  termOr = IntersectionInt.make(componentT, componentB, memory);
80  if (truthB.isNegative()) {
81  if (!truthT.isNegative()) {
82  termAnd = DifferenceExt.Companion.make(componentT,
83  componentB, memory);
84  truthAnd = TruthFunctions.intersection(truthT,
85  TruthFunctions.negation(truthB));
86  }
87  } else if (truthT.isNegative()) {
88  termAnd = DifferenceExt.Companion.make(componentB,
89  componentT, memory);
90  truthAnd = TruthFunctions.intersection(truthB,
91  TruthFunctions.negation(truthT));
92  } else {
93  termAnd = IntersectionExt.make(componentT, componentB,
94  memory);
95  }
96  } else if (taskContent instanceof Implication) {
97  termOr = Disjunction.Companion.make(componentT, componentB,
98  memory);
99  termAnd = Conjunction.Companion.make(componentT, componentB,
100  memory);
101  }
102  processComposed(taskContent, componentCommon, termOr, truthOr,
103  memory);
104  processComposed(taskContent, componentCommon, termAnd, truthAnd,
105  memory);
106  } else { // index == 1
107  if (taskContent instanceof Inheritance) {
108  termOr = IntersectionExt.make(componentT, componentB, memory);
109  if (truthB.isNegative()) {
110  if (!truthT.isNegative()) {
111  termAnd = DifferenceInt.Companion.make(componentT,
112  componentB, memory);
113  truthAnd = TruthFunctions.intersection(truthT,
114  TruthFunctions.negation(truthB));
115  }
116  } else if (truthT.isNegative()) {
117  termAnd = DifferenceInt.Companion.make(componentB,
118  componentT, memory);
119  truthAnd = TruthFunctions.intersection(truthB,
120  TruthFunctions.negation(truthT));
121  } else {
122  termAnd = IntersectionInt.make(componentT, componentB,
123  memory);
124  }
125  } else if (taskContent instanceof Implication) {
126  termOr = Conjunction.Companion.make(componentT, componentB,
127  memory);
128  termAnd = Disjunction.Companion.make(componentT, componentB,
129  memory);
130  }
131  processComposed(taskContent, termOr, componentCommon, truthOr,
132  memory);
133  processComposed(taskContent, termAnd, componentCommon, truthAnd,
134  memory);
135  }
136  if (taskContent instanceof Inheritance) {
137  introVarOuter(taskContent, beliefContent, index, memory);// introVarImage(taskContent,
138  // beliefContent,
139  // index,
140  // memory);
141  }
142  }

156  private static void processComposed(Statement statement, Term subject,
157  Term predicate, TruthValue truth, Memory memory) {
158  if (subject == null || predicate == null) {
159  return;
160  }

188  private static void decomposeCompound(CompoundTerm compound,
189  Term component, Term term1, int index, boolean compoundTask,
190  Memory memory) {
191  if (!(compound instanceof Statement)) {
192  Term term2 = CompoundTerm.reduceComponents(compound, component,
193  memory);
194  if (term2 != null) {
195  Task task = memory.getCurrentTask();
196  Sentence sentence = task.getSentence();
197  Sentence belief = memory.getCurrentBelief();
198  Statement oldContent = (Statement) task.getContent();
199  TruthValue v1, v2;
200  if (compoundTask) {
201  v1 = sentence.getTruth();
202  v2 = belief.getTruth();
203  } else {
204  v1 = belief.getTruth();
205  v2 = sentence.getTruth();
206  }
207  TruthValue truth = null;
208  Term content;
209  if (index == 0) {
210  content = Statement.make(oldContent, term1, term2, memory);
211  if (content != null && oldContent instanceof Inheritance) {
212  if (!(compound instanceof IntersectionExt)) {
213  if (!(compound instanceof IntersectionInt)) {
214  if (!(compound instanceof SetInt)
215  || !(component instanceof SetInt)) {
216  if (!(compound instanceof SetExt)
217  || !(component instanceof SetExt)) {
218  if (!(compound instanceof DifferenceExt)) {
219  if (!(oldContent instanceof Implication)) {
220  return;
221  }
222  if (compound instanceof Conjunction) {
223  truth = TruthFunctions
224  .reduceConjunction(v1,
225  v2);
226  } else if (compound instanceof Disjunction) {
227  truth = reduceDisjunction(v1,
228  v2);
229  }
230  } else {
231  if (compound.componentAt(0).equals(
232  component)) {
233  truth = reduceDisjunction(v2,
234  v1);
235  } else {
236  truth = reduceConjunctionNeg(
237  v1, v2);
238  }
239  }
240  } else {
241  truth = reduceDisjunction(v1, v2);
242  }
243  } else {
244  truth = TruthFunctions.reduceConjunction(
245  v1, v2);
246  }
247  } else {
248  truth = reduceDisjunction(v1, v2);
249  }
250  } else {
251  truth = TruthFunctions.reduceConjunction(v1, v2);
252  }
253  }
254  } else {
255  content = Statement.make(oldContent, term2, term1, memory);
256  if (content == null) {
257  return;
258  }
259  if (oldContent instanceof Inheritance) {
260  if (compound instanceof IntersectionInt) {
261  truth = TruthFunctions.reduceConjunction(v1, v2);
262  } else if (compound instanceof IntersectionExt) {
263  truth = reduceDisjunction(v1, v2);
264  } else if (compound instanceof SetExt
265  && component instanceof SetExt) {
266  truth = TruthFunctions.reduceConjunction(v1, v2);
267  } else if (compound instanceof SetInt
268  && component instanceof SetInt) {
269  truth = reduceDisjunction(v1, v2);
270  } else if (compound instanceof DifferenceInt) {
271  if (compound.componentAt(1).equals(component)) {
272  truth = reduceDisjunction(v2, v1);
273  } else {
274  truth = reduceConjunctionNeg(v1, v2);
275  }
276  }
277  } else if (oldContent instanceof Implication)
278  if (compound instanceof Disjunction) {
279  truth = TruthFunctions.reduceConjunction(v1, v2);
280  } else if (compound instanceof Conjunction) {
281  truth = reduceDisjunction(v1, v2);
282  }
283  }
284  if (truth != null) {
285  BudgetValue budget = BudgetFunctions.compoundForward(
286  memory, truth, content);
287  Memory.doublePremiseTask(memory, content, truth, budget);
288  }
289  }
290  }

308  private static void introVarOuter(Statement taskContent, Statement beliefContent, int index, Memory memory) {

355  Statement state2 = Inheritance.make(term21, term22, memory);

357  TruthValue truth = TruthFunctions.induction(truthT, truthB);

362  budget = BudgetFunctions.compoundForward(memory, truth, content);

365  truth = TruthFunctions.comparison(truthT, truthB);
366  budget = BudgetFunctions.compoundForward(memory, truth, content);
367  Memory.doublePremiseTask(memory, content, truth, budget);
368  Variable varDep = new Variable("#varDep");
369  if (index == 0) {
370  state1 = Inheritance.make(varDep, taskContent.getPredicate(), memory);
371  state2 = Inheritance.make(varDep, beliefContent.getPredicate(), memory);
372  } else {
373  state1 = Inheritance.make(taskContent.getSubject(), varDep, memory);
374  state2 = Inheritance.make(beliefContent.getSubject(), varDep, memory);
375  }
376  content = Conjunction.Companion.make(state1, state2, memory);
377  truth = TruthFunctions.intersection(truthT, truthB);
378  budget = BudgetFunctions.compoundForward(memory, truth, content);
379  Memory.doublePremiseTask(memory, content, truth, budget, false);
380  }

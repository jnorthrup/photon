92  int i1, i2, j;

94  baseLength = Math.min(first.getBaseLength() + second.getBaseLength(), Parameters.MAXIMUM_STAMP_LENGTH);

99  j++;

101  i2++;

105  getEvidentialBase().add(first.get(i1));

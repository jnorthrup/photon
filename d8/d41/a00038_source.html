217  private static void inferToSym(Sentence judgment1,
218  SentenceStruct judgment2, Memory memory) {
219  Statement s1 = (Statement) judgment1.getContent();
220  Term t1 = s1.getSubject();
221  Term t2 = s1.getPredicate();
222  Term content;
223  if (s1 instanceof Inheritance) {
224  content = Similarity.make(t1, t2, memory);
225  } else {
226  content = Equivalence.Companion.make(t1, t2, memory);
227  }
228  TruthValue value1 = judgment1.getTruth();
229  TruthHandle value2 = judgment2.getTruth();
230  TruthValue truth = TruthFunctions.intersection(value1, value2);
231  BudgetValue budget = BudgetFunctions.forward(memory, truth);
232  Memory.doublePremiseTask(memory, content, truth, budget);

340  static BudgetValue revise(TruthValue tTruth, TruthValue bTruth,
341  TruthValue truth, boolean feedbackToLinks, WorkSpace memory) {
342  float difT = truth.isExpDifAbs(tTruth);
343  Task task = memory.getCurrentTask();
344  Item.decPriority(task.getBudget(), 1 - difT);
345  Item.decDurability(task.getBudget(), 1 - difT);
346  if (feedbackToLinks) {
347  TaskLink tLink = memory.getCurrentTaskLink();
348  Item.decPriority(tLink.getBudget(), 1 - difT);
349  Item.decDurability(tLink.getBudget(), 1 - difT);
350  TermLink bLink = memory.getCurrentBeliefLink();
351  float difB = truth.isExpDifAbs(bTruth);
352  Item.decPriority(bLink.getBudget(), 1 - difB);
353  Item.decDurability(bLink.getBudget(), 1 - difB);
354  }
355  float dif = truth.getConfidence()
356  - Math.max(tTruth.getConfidence(), bTruth.getConfidence());
357  float priority = UtilityFunctions.or(dif, task.getPriority());
358  float durability = UtilityFunctions.or(dif, task.getDurability());
359  float quality = BudgetFunctions.truthToQuality(truth);
360  return new BudgetValue(priority, durability, quality);
361  }

377  static BudgetValue solutionEval(SentenceStruct problem,
378  SentenceStruct solution, Task task, WorkSpace memory) {
379  BudgetValue budget = null;
380  boolean feedbackToLinks = false;
381  if (task == null) { // called in continued processing
382  task = memory.getCurrentTask();
383  feedbackToLinks = true;
384  }
385  boolean judgmentTask = task.getSentence().isJudgment();
386  float quality = solutionQuality(problem, solution);
387  if (judgmentTask) {
388  Item.incPriority(task.getBudget(), quality);
389  } else {
390  task.setPriority(Math.min(1 - quality, task.getPriority()));
391  budget = new BudgetValue(quality, task.getDurability(),
392  BudgetFunctions.truthToQuality(solution.getTruth()));
393  }
394  if (feedbackToLinks) {
395  TaskLink tLink = memory.getCurrentTaskLink();
396  tLink.setPriority(Math.min(1 - quality, tLink.getPriority()));
397  TermLink bLink = memory.getCurrentBeliefLink();
398  Item.incPriority(bLink.getBudget(), quality);
399  }
400  return budget;
401  }
